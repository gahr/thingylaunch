CMAKE_MINIMUM_REQUIRED (VERSION 2.8)

SET (TL_PROJECT_NAME thingylaunch)
SET (TL_PROJECT_VERS 1.1.7)

PROJECT (${TL_PROJECT_NAME})

FIND_PACKAGE (X11)
IF (NOT X11_FOUND)
    MESSAGE (FATAL_ERROR "Could not locate X11.")
ENDIF ()

INCLUDE_DIRECTORIES (${X11_X11_INCLUDE_PATH})

ADD_EXECUTABLE (
    ${TL_PROJECT_NAME}
    bookmark.c
    completion.c
    history.c
    thingylaunch.c
)

TARGET_LINK_LIBRARIES (
    ${TL_PROJECT_NAME}
    ${X11_LIBRARIES}
)

INSTALL (
    TARGETS ${TL_PROJECT_NAME}
    RUNTIME DESTINATION bin
)
